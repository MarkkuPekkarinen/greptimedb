50#[async_trait]
51impl OpenTelemetryProtocolHandler for Instance {
52    #[tracing::instrument(skip_all)]
53    async fn metrics(
54        &self,
55        request: ExportMetricsServiceRequest,
56        ctx: QueryContextRef,
57    ) -> ServerResult<Output> {
58        self.plugins
59            .get::<PermissionCheckerRef>()
60            .as_ref()
61            .check_permission(ctx.current_user(), PermissionReq::Otlp)
62            .context(AuthSnafu)?;
63
64        let interceptor_ref = self
65            .plugins
66            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
67        interceptor_ref.pre_execute(ctx.clone())?;
68
69        let input_names = request
70            .resource_metrics
71            .iter()
72            .flat_map(|r| r.scope_metrics.iter())
73            .flat_map(|s| s.metrics.iter().map(|m| &m.name))
74            .collect::<Vec<_>>();
75
76        // See [`OtlpMetricCtx`] for details
77        let is_legacy = self.check_otlp_legacy(&input_names, ctx.clone()).await?;
78
79        let mut metric_ctx = ctx
80            .protocol_ctx()
81            .get_otlp_metric_ctx()
82            .cloned()
83            .unwrap_or_default();
84        metric_ctx.is_legacy = is_legacy;
85
86        let (requests, rows, semantic_index) =
87            otlp::metrics::to_grpc_insert_requests(request, &mut metric_ctx)?;
88        OTLP_METRICS_ROWS.inc_by(rows as u64);
89
90        let ctx = {
91            let mut c = (*ctx).clone();
92            c.set_extension(SEMANTIC_SIGNAL_TYPE, SIGNAL_TYPE_METRIC);
93            c.set_extension(SEMANTIC_SOURCE, SOURCE_OPENTELEMETRY);
94            // Per-table metric specifics + resource/scope lineage ride this
95            // internal channel; the auto-create path folds them per table name.
96            if let Some(index) = semantic_index.encode() {
97                c.set_extension(SEMANTIC_PER_TABLE_INDEX_KEY, index);
98            }
99            if !is_legacy {
100                c.set_extension(OTLP_METRIC_COMPAT_KEY, OTLP_METRIC_COMPAT_PROM.to_string());
101            }
102            Arc::new(c)
103        };
104
105        // If the user uses the legacy path, it is by default without metric engine.
106        if metric_ctx.is_legacy || !metric_ctx.with_metric_engine {
107            self.handle_row_inserts(requests, ctx, false, false)
108                .await
109                .map_err(BoxedError::new)
110                .context(error::ExecuteGrpcQuerySnafu)
111        } else {
112            let physical_table = ctx
113                .extension(PHYSICAL_TABLE_PARAM)
114                .unwrap_or(GREPTIME_PHYSICAL_TABLE)
115                .to_string();
116            self.handle_metric_row_inserts(requests, ctx, physical_table.clone())
117                .await
118                .map_err(BoxedError::new)
119                .context(error::ExecuteGrpcQuerySnafu)
120        }
121    }

123    #[tracing::instrument(skip_all)]
124    async fn traces(
125        &self,
126        pipeline_handler: PipelineHandlerRef,
127        request: ExportTraceServiceRequest,
128        pipeline: PipelineWay,
129        pipeline_params: GreptimePipelineParams,
130        table_name: String,
131        ctx: QueryContextRef,
132    ) -> ServerResult<TraceIngestOutcome> {
133        self.plugins
134            .get::<PermissionCheckerRef>()
135            .as_ref()
136            .check_permission(ctx.current_user(), PermissionReq::Otlp)
137            .context(AuthSnafu)?;
138
139        let interceptor_ref = self
140            .plugins
141            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
142        interceptor_ref.pre_execute(ctx.clone())?;
143
144        // `schema_url` is consumed by `parse`, so derive conventions first.
145        let conventions = trace_conventions(&request);
146        let spans = otlp::trace::span::parse(request);
147        self.ingest_trace_spans(
148            pipeline_handler,
149            &pipeline,
150            &pipeline_params,
151            table_name,
152            spans,
153            &conventions,
154            ctx,
155        )
156        .await
157    }
158
159    #[tracing::instrument(skip_all)]
160    async fn logs(
161        &self,
162        pipeline_handler: PipelineHandlerRef,
163        request: ExportLogsServiceRequest,
164        pipeline: PipelineWay,
165        pipeline_params: GreptimePipelineParams,
166        table_name: String,
167        ctx: QueryContextRef,
168    ) -> ServerResult<Vec<Output>> {
169        self.plugins
170            .get::<PermissionCheckerRef>()
171            .as_ref()
172            .check_permission(ctx.current_user(), PermissionReq::Otlp)
173            .context(AuthSnafu)?;
174
175        let interceptor_ref = self
176            .plugins
177            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
178        interceptor_ref.pre_execute(ctx.clone())?;
179
180        // `as_req_iter` clones this ctx into each `temp_ctx`, so identity set here
181        // reaches the context that drives table auto-create.
182        let ctx = {
183            let mut c = (*ctx).clone();
184            c.set_extension(SEMANTIC_SIGNAL_TYPE, SIGNAL_TYPE_LOG);
185            c.set_extension(SEMANTIC_SOURCE, SOURCE_OPENTELEMETRY);
186            Arc::new(c)
187        };
188
189        let opt_req = otlp::logs::to_grpc_insert_requests(
190            request,
191            pipeline,
192            pipeline_params,
193            table_name,
194            &ctx,
195            pipeline_handler,
196        )
197        .await?;
198
199        let mut outputs = vec![];
200
201        for (temp_ctx, requests) in opt_req.as_req_iter(ctx) {
202            let cnt = requests
203                .inserts
204                .iter()
205                .filter_map(|r| r.rows.as_ref().map(|r| r.rows.len()))
206                .sum::<usize>();
207
208            let o = self
209                .handle_log_inserts(requests, temp_ctx)
210                .await
211                .inspect(|_| OTLP_LOGS_ROWS.inc_by(cnt as u64))
212                .map_err(BoxedError::new)
213                .context(error::ExecuteGrpcQuerySnafu)?;
214            outputs.push(o);
215        }
216
217        Ok(outputs)
218    }
