47/// Distinguishes raw bytes from JSONB values that both use a binary protobuf value.
48#[derive(Debug, Clone, Copy, PartialEq, Eq)]
49pub enum TraceBinaryType {
50    /// Raw OTLP bytes with no datatype extension.
51    Binary,
52    /// An OTLP array or key-value list encoded with the JSONB extension.
53    Json,
54}
55
56impl TraceBinaryType {
57    /// Applies this logical binary type to a column schema.
58    pub fn apply_to_schema(self, schema: &mut ColumnSchema) {
59        schema.datatype = ColumnDataType::Binary as i32;
60        schema.datatype_extension = match self {
61            Self::Binary => None,
62            Self::Json => Some(ColumnDataTypeExtension {
63                type_ext: Some(TypeExt::JsonType(JsonTypeExtension::JsonBinary.into())),
64            }),
65        };
66    }
67}

69/// Per-column observations collected while building one trace chunk.
70#[derive(Default)]
71struct TraceBatchColumnSchema {
72    value_types: Vec<ColumnDataType>,
73    present_rows: Vec<usize>,
74    binary_types: Vec<(usize, TraceBinaryType)>,
75}
76
77impl TraceBatchColumnSchema {
78    /// Records a row once while preserving encounter order.
79    fn observe_row(&mut self, row_index: usize) {
80        if self.present_rows.last() != Some(&row_index) {
81            self.present_rows.push(row_index);
82        }
83    }
84
85    /// Returns whether raw binary and JSONB values share this column.
86    fn has_incompatible_logical_types(&self) -> bool {
87        let Some((_, first_type)) = self.binary_types.first() else {
88            return false;
89        };
90        self.binary_types
91            .iter()
92            .any(|(_, binary_type)| binary_type != first_type)
93    }
94}

96/// Sparse column metadata retained for span-level fallback.
97pub struct TraceRetryColumn {
98    /// Row indexes that contain this dynamic column.
99    pub present_rows: Vec<usize>,
100    /// Row-specific binary kinds needed to restore the logical schema.
101    pub binary_types: Vec<(usize, TraceBinaryType)>,
102}
103
104/// Retry metadata keyed by dynamic trace column name.
105pub type TraceRetryColumns = HashMap<String, TraceRetryColumn>;
106
107/// Schema observations for dynamic columns in one converted trace chunk.
108#[derive(Default)]
109pub struct TraceBatchSchema {
110    columns: HashMap<String, TraceBatchColumnSchema>,
111}
112
113impl TraceBatchSchema {
114    /// Records a scalar value type and the row where the column is present.
115    fn observe_value_type(&mut self, name: &str, row_index: usize, value_type: ColumnDataType) {
116        let column = self.columns.entry(name.to_string()).or_default();
117        column.observe_row(row_index);
118        if !column.value_types.contains(&value_type) {
119            column.value_types.push(value_type);
120        }
121    }
122
123    /// Records the logical kind of a binary protobuf value for one row.
124    fn observe_binary_type(&mut self, name: &str, row_index: usize, binary_type: TraceBinaryType) {
125        let column = self.columns.entry(name.to_string()).or_default();
126        column.observe_row(row_index);
127        if !column.value_types.contains(&ColumnDataType::Binary) {
128            column.value_types.push(ColumnDataType::Binary);
129        }
130        if let Some((last_row_index, last_type)) = column.binary_types.last_mut()
131            && *last_row_index == row_index
132        {
133            *last_type = binary_type;
134        } else {
135            column.binary_types.push((row_index, binary_type));
136        }
137    }
138
139    /// Records a sparse column occurrence that has no dynamic value type.
140    fn observe_present_column(&mut self, name: &str, row_index: usize) {
141        self.columns
142            .entry(name.to_string())
143            .or_default()
144            .observe_row(row_index);
145    }
146
147    /// Returns the distinct value types in first-seen order for a column.
148    pub fn value_types(&self, name: &str) -> Option<&[ColumnDataType]> {
149        self.columns
150            .get(name)
151            .map(|column| column.value_types.as_slice())
152    }
153
154    /// Returns whether any column mixes raw binary and JSONB values.
155    pub fn has_incompatible_logical_types(&self) -> bool {
156        self.columns
157            .values()
158            .any(TraceBatchColumnSchema::has_incompatible_logical_types)
159    }
160
161    /// Returns whether the named column mixes raw binary and JSONB values.
162    pub fn has_incompatible_logical_types_for(&self, name: &str) -> bool {
163        self.columns
164            .get(name)
165            .is_some_and(TraceBatchColumnSchema::has_incompatible_logical_types)
166    }
167
168    /// Converts batch observations into sparse metadata for per-span retries.
169    pub fn into_retry_columns(self) -> TraceRetryColumns {
170        self.columns
171            .into_iter()
172            .filter(|(_, column)| !column.present_rows.is_empty())
173            .map(|(name, column)| {
174                let binary_types = if column.has_incompatible_logical_types()
175                    || column
176                        .value_types
177                        .iter()
178                        .any(|datatype| *datatype != ColumnDataType::Binary)
179                {
180                    column.binary_types
181                } else {
182                    Vec::new()
183                };
184                (
185                    name,
186                    TraceRetryColumn {
187                        present_rows: column.present_rows,
188                        binary_types,
189                    },
190                )
191            })
192            .collect()
193    }
194}

196/// Converts trace spans into row insert requests for the main v1 trace table.
197///
198/// Auxiliary service and operation table writes are built separately so the
199/// caller can update them only after the main span write succeeds.
200pub fn v1_to_grpc_main_insert_requests(
201    spans: &[TraceSpan],
202    _pipeline: &PipelineWay,
203    _pipeline_params: &GreptimePipelineParams,
204    table_name: &str,
205    _query_ctx: &QueryContextRef,
206    _pipeline_handler: PipelineHandlerRef,
207) -> Result<(RowInsertRequests, usize)> {
208    let requests = v1_to_grpc_main_insert_requests_from_iter(spans.iter().cloned(), table_name)?;
209    Ok((requests, spans.len()))
210}

219/// Builds the main-table request without collecting batch schema observations.
220fn v1_to_grpc_main_insert_requests_from_iter(
221    spans: impl ExactSizeIterator<Item = TraceSpan>,
222    table_name: &str,
223) -> Result<RowInsertRequests> {
224    let mut multi_table_writer = MultiTableData::default();
225    let trace_writer = build_trace_table_data_from_iter(spans, None)?;
226    multi_table_writer.add_table_data(table_name, trace_writer);
227
228    Ok(multi_table_writer.into_row_insert_requests().0)
229}

231/// Builds the row-oriented payload for the main v1 trace table.
232pub fn build_trace_table_data(spans: &[TraceSpan]) -> Result<TableData> {
233    build_trace_table_data_from_iter(spans.iter().cloned(), None)
234}

236/// Builds trace rows while collecting dynamic column observations.
237fn build_trace_table_data_with_schema(
238    spans: impl ExactSizeIterator<Item = TraceSpan>,
239) -> Result<(TableData, TraceBatchSchema)> {
240    let mut batch_schema = TraceBatchSchema::default();
241    let trace_writer = build_trace_table_data_from_iter(spans, Some(&mut batch_schema))?;
242    Ok((trace_writer, batch_schema))
243}
244
245/// Shared row builder with optional batch schema observation.
246fn build_trace_table_data_from_iter(
247    spans: impl ExactSizeIterator<Item = TraceSpan>,
248    mut batch_schema: Option<&mut TraceBatchSchema>,
249) -> Result<TableData> {
250    let mut trace_writer = TableData::new(APPROXIMATE_COLUMN_COUNT, spans.len());
251    for span in spans {
252        let row_index = trace_writer.num_rows();
253        write_span_to_row_inner(
254            &mut trace_writer,
255            span,
256            row_index,
257            batch_schema.as_deref_mut(),
258        )?;
259    }
260    Ok(trace_writer)
261}

263/// Builds row insert requests for the v1 trace auxiliary tables.
264pub fn build_aux_table_requests(
265    aux_data: TraceAuxData,
266    table_name: &str,
267) -> Result<(RowInsertRequests, usize)> {
268    let mut multi_table_writer = MultiTableData::default();
269    let mut trace_services_writer = TableData::new(APPROXIMATE_COLUMN_COUNT, 1);
270    let mut trace_operations_writer = TableData::new(APPROXIMATE_COLUMN_COUNT, 1);
271
272    write_trace_services_to_row(&mut trace_services_writer, aux_data.services)?;
273    write_trace_operations_to_row(&mut trace_operations_writer, aux_data.operations)?;
274
275    multi_table_writer.add_table_data(trace_services_table_name(table_name), trace_services_writer);
276    multi_table_writer.add_table_data(
277        trace_operations_table_name(table_name),
278        trace_operations_writer,
279    );
280
281    Ok(multi_table_writer.into_row_insert_requests())
282}

284pub fn write_span_to_row(writer: &mut TableData, span: TraceSpan) -> Result<()> {
285    let row_index = writer.num_rows();
286    write_span_to_row_inner(writer, span, row_index, None)
287}

289/// Writes one span and optionally records its dynamic columns for reconciliation.
290fn write_span_to_row_inner(
291    writer: &mut TableData,
292    span: TraceSpan,
293    row_index: usize,
294    mut batch_schema: Option<&mut TraceBatchSchema>,
295) -> Result<()> {
296    let mut row = writer.alloc_one_row();
297
298    // write ts
299    row_writer::write_ts_to_nanos(
300        writer,
301        TIMESTAMP_COLUMN,
302        Some(span.start_in_nanosecond as i64),
303        Precision::Nanosecond,
304        &mut row,
305    )?;
306
307    // write fields
308    let fields = vec![
309        make_column_data(
310            "timestamp_end",
311            ColumnDataType::TimestampNanosecond,
312            Some(ValueData::TimestampNanosecondValue(
313                span.end_in_nanosecond as i64,
314            )),
315        ),
316        make_column_data(
317            DURATION_NANO_COLUMN,
318            ColumnDataType::Uint64,
319            Some(ValueData::U64Value(
320                span.end_in_nanosecond - span.start_in_nanosecond,
321            )),
322        ),
323        make_string_column_data(PARENT_SPAN_ID_COLUMN, span.parent_span_id),
324        make_string_column_data(TRACE_ID_COLUMN, Some(span.trace_id)),
325        make_string_column_data(SPAN_ID_COLUMN, Some(span.span_id)),
326        make_string_column_data(SPAN_KIND_COLUMN, Some(span.span_kind)),
327        make_string_column_data(SPAN_NAME_COLUMN, Some(span.span_name)),
328        make_string_column_data(SPAN_STATUS_CODE, Some(span.span_status_code)),
329        make_string_column_data(SPAN_STATUS_MESSAGE_COLUMN, Some(span.span_status_message)),
330        make_string_column_data(TRACE_STATE_COLUMN, Some(span.trace_state)),
331        make_string_column_data(SCOPE_NAME_COLUMN, Some(span.scope_name)),
332        make_string_column_data(SCOPE_VERSION_COLUMN, Some(span.scope_version)),
333    ];
334    row_writer::write_fields(writer, fields.into_iter(), &mut row)?;
335
336    if let Some(service_name) = span.service_name {
337        if let Some(batch_schema) = batch_schema.as_deref_mut() {
338            batch_schema.observe_present_column(SERVICE_NAME_COLUMN, row_index);
339        }
340        row_writer::write_tags(
341            writer,
342            std::iter::once((SERVICE_NAME_COLUMN.to_string(), service_name)),
343            &mut row,
344        )?;
345    }
346
347    write_attributes_with_schema(
348        writer,
349        "span_attributes",
350        span.span_attributes,
351        &mut row,
352        row_index,
353        batch_schema.as_deref_mut(),
354    )?;
355    write_attributes_with_schema(
356        writer,
357        "scope_attributes",
358        span.scope_attributes,
359        &mut row,
360        row_index,
361        batch_schema.as_deref_mut(),
362    )?;
363    write_attributes_with_schema(
364        writer,
365        "resource_attributes",
366        span.resource_attributes,
367        &mut row,
368        row_index,
369        batch_schema,
370    )?;
371
372    row_writer::write_json(
373        writer,
374        SPAN_EVENTS_COLUMN,
375        span.span_events.into(),
376        &mut row,
377    )?;
378    row_writer::write_json(writer, "span_links", span.span_links.into(), &mut row)?;
379
380    writer.add_row(row);
381
382    Ok(())
383}

385fn write_trace_services_to_row(writer: &mut TableData, services: HashSet<String>) -> Result<()> {
386    for service_name in services {
387        let mut row = writer.alloc_one_row();
388        // Write the timestamp as 0.
389        row_writer::write_ts_to_nanos(
390            writer,
391            TIMESTAMP_COLUMN,
392            Some(MAX_TIMESTAMP),
393            Precision::Nanosecond,
394            &mut row,
395        )?;
396
397        // Write the `service_name` column.
398        row_writer::write_tags(
399            writer,
400            std::iter::once((SERVICE_NAME_COLUMN.to_string(), service_name)),
401            &mut row,
402        )?;
403        writer.add_row(row);
404    }
405
406    Ok(())
407}

409fn write_trace_operations_to_row(
410    writer: &mut TableData,
411    operations: HashSet<(String, String, String)>,
412) -> Result<()> {
413    for (service_name, span_name, span_kind) in operations {
414        let mut row = writer.alloc_one_row();
415        // Write the timestamp as 0.
416        row_writer::write_ts_to_nanos(
417            writer,
418            TIMESTAMP_COLUMN,
419            Some(MAX_TIMESTAMP),
420            Precision::Nanosecond,
421            &mut row,
422        )?;
423
424        // Write the `service_name`, `span_name`, and `span_kind` columns as tags.
425        row_writer::write_tags(
426            writer,
427            vec![
428                (SERVICE_NAME_COLUMN.to_string(), service_name),
429                (SPAN_NAME_COLUMN.to_string(), span_name),
430                (SPAN_KIND_COLUMN.to_string(), span_kind),
431            ]
432            .into_iter(),
433            &mut row,
434        )?;
435        writer.add_row(row);
436    }
437
438    Ok(())
439}

442pub(crate) fn write_attributes(
443    writer: &mut TableData,
444    prefix: &str,
445    attributes: Attributes,
446    row: &mut Vec<Value>,
447) -> Result<()> {
448    let row_index = writer.num_rows();
449    write_attributes_with_schema(writer, prefix, attributes, row, row_index, None)

452/// Writes flattened attributes without coercion and optionally records their actual types.
453fn write_attributes_with_schema(
454    writer: &mut TableData,
455    prefix: &str,
456    attributes: Attributes,
457    row: &mut Vec<Value>,
458    row_index: usize,
459    mut batch_schema: Option<&mut TraceBatchSchema>,
460) -> Result<()> {
461    for attr in attributes.take().into_iter() {
462        let key_suffix = attr.key;
463        // skip resource_attributes.service.name because its already copied to
464        // top level as `SERVICE_NAME_COLUMN`
465        if prefix == "resource_attributes" && key_suffix == KEY_SERVICE_NAME {
466            continue;
467        }
468
469        let key = format!("{}.{}", prefix, key_suffix);
470        match attr.value.and_then(|v| v.value) {
471            Some(OtlpValue::StringValue(v)) => {
472                if let Some(batch_schema) = batch_schema.as_deref_mut() {
473                    batch_schema.observe_value_type(&key, row_index, ColumnDataType::String);
474                }
475                // Keep the raw request value here. Mixed trace types are reconciled later
476                // in the frontend once we can also see the existing table schema.
477                writer.write_field_unchecked(
478                    &key,
479                    ColumnDataType::String,
480                    Some(ValueData::StringValue(v)),
481                    row,
482                );
483            }
484            Some(OtlpValue::BoolValue(v)) => {
485                if let Some(batch_schema) = batch_schema.as_deref_mut() {
486                    batch_schema.observe_value_type(&key, row_index, ColumnDataType::Boolean);
487                }
488                // Do not coerce or promote types while building the request-local rows.
489                writer.write_field_unchecked(
490                    &key,
491                    ColumnDataType::Boolean,
492                    Some(ValueData::BoolValue(v)),
493                    row,
494                );
495            }
496            Some(OtlpValue::IntValue(v)) => {
497                if let Some(batch_schema) = batch_schema.as_deref_mut() {
498                    batch_schema.observe_value_type(&key, row_index, ColumnDataType::Int64);
499                }
500                // Preserving the original value avoids order-dependent behavior inside one batch.
501                writer.write_field_unchecked(
502                    &key,
503                    ColumnDataType::Int64,
504                    Some(ValueData::I64Value(v)),
505                    row,
506                );
507            }
508            Some(OtlpValue::DoubleValue(v)) => {
509                if let Some(batch_schema) = batch_schema.as_deref_mut() {
510                    batch_schema.observe_value_type(&key, row_index, ColumnDataType::Float64);
511                }
512                writer.write_field_unchecked(
513                    &key,
514                    ColumnDataType::Float64,
515                    Some(ValueData::F64Value(v)),
516                    row,
517                );
518            }
519            Some(OtlpValue::ArrayValue(v)) => {
520                if let Some(batch_schema) = batch_schema.as_deref_mut() {
521                    batch_schema.observe_binary_type(&key, row_index, TraceBinaryType::Json);
522                }
523                writer.write_column_unchecked(
524                    row_writer::build_json_column_schema(key),
525                    Some(ValueData::BinaryValue(
526                        any_value_to_jsonb(OtlpValue::ArrayValue(v)).to_vec(),
527                    )),
528                    row,
529                );
530            }
531            Some(OtlpValue::KvlistValue(v)) => {
532                if let Some(batch_schema) = batch_schema.as_deref_mut() {
533                    batch_schema.observe_binary_type(&key, row_index, TraceBinaryType::Json);
534                }
535                writer.write_column_unchecked(
536                    row_writer::build_json_column_schema(key),
537                    Some(ValueData::BinaryValue(
538                        any_value_to_jsonb(OtlpValue::KvlistValue(v)).to_vec(),
539                    )),
540                    row,
541                );
542            }
543            Some(OtlpValue::BytesValue(v)) => {
544                if let Some(batch_schema) = batch_schema.as_deref_mut() {
545                    batch_schema.observe_binary_type(&key, row_index, TraceBinaryType::Binary);
546                }
547                writer.write_field_unchecked(
548                    key,
549                    ColumnDataType::Binary,
550                    Some(ValueData::BinaryValue(v)),
551                    row,
552                );
553            }
554            None => {}
555        }
556    }
557
558    Ok(())
559}
